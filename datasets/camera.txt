import cv2

cap = cv2.VideoCapture(0)
cap.set(cv2.CAP_PROP_FRAME_WIDTH, 3840)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 2160)

i=1
name = 'dot'
while(1):
  ret ,frame = cap.read()
  k=cv2.waitKey(1)
  if k==27:
     break
  elif k==ord('s'):
     cv2.imwrite('C:/Users/st313/Desktop/canp/dot/'+name+str(i)+'.jpg',frame)
     i+=1
  cv2.imshow("capture", frame)

cap.release()
cv2.destroyAllwindows()
